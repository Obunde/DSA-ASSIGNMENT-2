#include <stdio.h>
#include <stdbool.h>

struct Node {
    int data;
    struct Node *next;
};

bool hasCycle(struct Node *head) {
    if (!head || !head->next)
        return false;
    
    struct Node *slow_ptr = head;
    struct Node *fast_ptr = head->next;
    
    while (slow_ptr != fast_ptr) {
        if (!fast_ptr || !fast_ptr->next)
            return false;
        slow_ptr = slow_ptr->next;
        fast_ptr = fast_ptr->next->next;
    }
    
    return true;
}
